IF db_id('BookStore') IS NULL
    CREATE DATABASE BookStore

GO

CREATE TABLE Author (
	AuthorID INTEGER PRIMARY KEY NOT NULL,
	AuthorFirstName VARCHAR(50) NOT NULL,
	AuthorLastName VARCHAR(50) NOT NULL,
	AuthorBirthYear DATETIME NOT NULL);

CREATE TABLE Book (
	BookID INTEGER PRIMARY KEY NOT NULL,
	BookName VARCHAR(50) NOT NULL,
	ReleaseYear DATETIME NOT NULL,
	AuthorID INTEGER NOT NULL,
	FOREIGN KEY (AuthorID) REFERENCES Author(AuthorID));

CREATE TABLE Stock (
	BookID INTEGER NOT NULL,
	FOREIGN KEY (BookID) REFERENCES Book(BookID),
	StockCount INTEGER NOT NULL,
	LoanLength INTEGER NOT NULL,
	StockID INTEGER PRIMARY KEY NOT NULL);
